  def new_tag_attributes=(tag_attributes)
    new_tags = Set.new

    tag_attributes.each do |a|
      new_tags.add(a)
    end

    new_tags.each do |nt|
      if nt[:content].eql? ""
        #do nothing
      else
        tag = Tag.find_or_create_by_content(nt[:content])
        if !tags.exists?(tag)
          tags << tag #self.tags
        end
      end
    end
  end
