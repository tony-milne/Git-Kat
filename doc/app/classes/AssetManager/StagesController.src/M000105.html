  def show
    @stage = Stage.find(params[:id])
    @assets = @stage.assets.paginate :per_page => 12, :page => params[:page],
                                     :conditions => ['title like ?', "%#{params[:search]}%"],
                                     :order => 'title'
    respond_to do |format|
      format.html # show.html.erb
      format.xml  { render :xml => @stage }
    end
  end
