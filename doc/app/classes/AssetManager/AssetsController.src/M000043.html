  def select_stage
    if !params[:stage][:id].empty?
      session[:stage_id] = params[:stage][:id].to_i
      redirect_to :back, :notice => "You can now add assets to #{Stage.find(session[:stage_id]).title}"
    else
      redirect_to :back, :notice => "No stage was selected"
    end
  end
